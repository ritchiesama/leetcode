class Solution {
    public int findMin(int[] nums) {
        int start = 0, end = nums.length - 1, mid;
        while(start < end){
            mid = start + (end - start) / 2;
            if(nums[mid] < nums[end]){
                end = mid;
            }else if(nums[mid] > nums[end]){
                start = mid + 1;
            }else if(nums[start] < nums[end]){
                return nums[start];
            }else{
                ++start;
            }
        }
        return nums[start];
    }
}
