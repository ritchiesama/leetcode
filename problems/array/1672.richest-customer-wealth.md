https://leetcode.com/problems/richest-customer-wealth/
```java
class Solution {
    public int maximumWealth(int[][] accounts) {
        int max = 0;
        int ith = accounts.length;
        int jth = accounts[0].length;
        for(int i = 0; i < ith; i++){
            int tmp = 0;
            for(int j = 0; j< jth; j++){
                tmp += accounts[i][j];
                }
            max = max > tmp ? max : tmp;
            }
        return max;
    }

}
```
