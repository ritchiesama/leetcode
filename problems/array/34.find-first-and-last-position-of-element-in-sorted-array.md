class Solution {
    public int[] searchRange(int[] nums, int target) {
        int[] current = new int[2];
        current[0] = -1;
        current[1] = -1;
        int count = 0;
        for(int i = 0; i<nums.length; i++){
            if(nums[i]== target){
                if(count == 0){
                current[0] = i;
                count++;
                }
                
                current[1] = i;
            }
        }
        return current;
    }
}
