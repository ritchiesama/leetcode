class Solution {
    public int maxProfit(int[] p) {
        int min = Integer.MAX_VALUE;
        int profit = 0;
        for(int i = 0; i< p.length; i++){
            int earn = p[i] - min;
            profit = Math.max(profit, earn);
            min = Math.min(min, p[i]);
        }
        return profit;
    }
}
