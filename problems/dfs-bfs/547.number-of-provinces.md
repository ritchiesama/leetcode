class Solution {
    public int findCircleNum(int[][] a) {
        int n = a.length, count = 0;
        boolean[] visited = new boolean[n];
        for(int i = 0; i < n; ++i){
            if(!visited[i]){
                dfs(a, i, visited);
                ++ count;
            }
            
        }
        return count;
    }
    private void dfs(int[][] a, int i, boolean[] v){
        v[i] = true;
        for(int k = 0; k < a.length; ++k ){
            if(a[i][k] == 1 && !v[k]){
                dfs(a, k, v);
            }
        }
    }
        
}
